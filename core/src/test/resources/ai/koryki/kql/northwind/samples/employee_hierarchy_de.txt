WITH roots AS (
 FIND mitarbeiter e
 FILTER NOT 
  EXISTS (e - berichtet_an mitarbeiter r
  )
 FETCH e.vorname fname, e.nachname lname, to_text(NULL) boss, to_int(1) lvl, lpad(to_text(e.mitarbeiter_id), 6, '0') path
 UNIONALL
 FIND mitarbeiter e, e - berichtet_an roots r
 FETCH e.vorname fname, e.nachname lname, r.lname boss, to_int(r.lvl + 1) lvl, concat(r.path, '/', lpad(to_text(e.mitarbeiter_id), 6, '0')) path
)
FIND roots r2
FETCH r2.lvl, concat(rpad(' ', r2.lvl * 2, ' '), r2.fname, ' ', r2.lname) hierarchy, r2.path ASC
